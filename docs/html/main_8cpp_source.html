102
103 for (int i = 0; lineas[i] != '\0'; i++) {
104 if (lineas[i] == ',') {

110 // Validar formato de trama

112 std::cout << "[ERROR] Formato de trama invalido: " << lineas << std::endl;
113 return;
114 }

116 // Extraer comando y dato
117 char comando = lineas[0];
118 const char* dato = &lineas[posicionComa + 1];

120 // Interpretar según el tipo de comando (polimorfismo)
121 switch(comando) {
122 case 'L':
123 case 'l':
124 {
125 // Trama LOAD: cargar un carácter
126 char letra;
127 if (strcmp(dato, "Space") == 0 || strcmp(dato, "space") == 0) {
128 letra = ' ';

130 letra = dato[0];
131 }
132
133 TramaBase* trama = new TramaLoad(letra);
134 trama->procesar(&ListaCarga, &RotorMapeo);
135 delete trama;
136 }
137 break;
138
139 case 'M':
140 case 'm':
141 {
142 // Trama MAP: rotar el rotor

145 trama->procesar(&ListaCarga, &RotorMapeo);
146 delete trama;
147 }
148 break;
149
